package step1;

public class Main {
	public static void main(String[] args) {


		/*
		 
*********
 ******* 
  *****  
   ***   
    *    
   ***   
  *****  
 ******* 
*********

		 */
		int i, j, k, p;
		k = 9;
		p = -1;

		for(i=0; i<9; i++) {
			if(i <=4) {
				k--;
				p++;
			}
			else {
				k++;
				p--;
			}
			for(j=0; j<9; j++) {
				if((p<=j) && (j<=k)) {
					System.out.print("*");
				}
				else  {
					System.out.print(" "); 
				}

			}
			System.out.println();
		}

	} 
}
